func (service *Service) getPlaylistSongs(playlistID string) error {

	var songsIds []string

	if err := service.context.Db.Select(&songsIds, `
				SELECT song_id FROM public.playlist_songs
				WHERE playlist_id = $1 `,
		playlistID); err != nil {
		return fmt.Errorf("Failed to get playlist songs, error: %s", err)
	}

}
